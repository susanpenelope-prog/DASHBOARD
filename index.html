function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); // ── Hoja Smart Flujo ── const sf = ss.getSheetByName("Smart Flujo") || ss.getSheetByName("Flujo Smart") || ss.getSheets()[0]; const sfData = sf.getDataRange().getValues(); // Fila 2 (índice 1): saldo inicial en col B (índice 1) const saldoInicial = parseFloat( String(sfData[1][1]).replace(/[^0-9.-]/g,'') ) || 0; // Filas de flujo desde fila 6 (índice 5) // Cols: A=Fecha(0) B=Tipo(1) C=Cat(2) D=Desc(3) E=Monto(4) F=Estado(5) const flujoRows = []; let saldoReal = saldoInicial; let saldoProyectado = saldoInicial; for (let i = 5; i < sfData.length; i++) { const row = sfData[i]; const tipo = String(row[1]||'').trim(); const monto = parseFloat(String(row[4]||0).replace(/[^0-9.-]/g,'')) || 0; const estado = String(row[5]||'').trim(); if (!tipo || monto === 0) continue; // I2: saldo real (solo Realizados) if (estado === 'Realizado') { if (tipo === 'Ingreso') saldoReal += monto; if (tipo === 'Egreso') saldoReal -= monto; } // Saldo proyectado (todos) if (tipo === 'Ingreso') saldoProyectado += monto; if (tipo === 'Egreso') saldoProyectado -= monto; flujoRows.push({ fecha: row[0] ? Utilities.formatDate(new Date(row[0]), 'America/Santiago', 'dd/MM/yyyy') : '', tipo, cat: String(row[2]||''), desc: String(row[3]||''), monto, estado, cot: String(row[6]||'') }); } // ── Hoja Master/Márgenes ── const mSheet = ss.getSheetByName("Master Operaciones & Márgenes") || ss.getSheetByName("Márgenes") || ss.getSheets()[1]; const mData = mSheet ? mSheet.getDataRange().getValues() : []; const obras = []; let headerIdx = -1; for (let i = 0; i < Math.min(mData.length,10); i++) { if (String(mData[i][0]).toLowerCase().includes('cotiz') || String(mData[i][1]).toLowerCase().includes('cliente')) { headerIdx = i; break; } } if (headerIdx >= 0) { for (let i = headerIdx+1; i < mData.length; i++) { const r = mData[i]; const cot = String(r[0]||'').trim(); const bruto = parseFloat(String(r[2]||0).replace(/[^0-9.-]/g,'')) || 0; if (!cot || bruto === 0) continue; obras.push({ cot, cliente: String(r[1]||''), bruto, costos: parseFloat(String(r[3]||0).replace(/[^0-9.-]/g,''))||0, neto: parseFloat(String(r[4]||0).replace(/[^0-9.-]/g,''))||0, margen: parseFloat(String(r[5]||0).replace(/[^0-9%]/g,''))||0 }); } } const result = { saldoInicial, saldoReal, saldoProyectado, ingresosPorCobrar: parseFloat( String(sfData[1][3]||0).replace(/[^0-9.-]/g,'') ) || 0, flujo: flujoRows, obras }; return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
